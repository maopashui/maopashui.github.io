Git 小记

2022-04-13 2 min read # git
常用命令
# 添加到缓存区
git add .

# 提交注释
git commit -m " comment "

# 提交到远程仓库
git push 

# 拉取仓库代码
git clone http://xxxx.git

# 同步远程仓库代码
git pull

# 查询当前.git指向的远程仓库地址
git remote -v

# 删除/添加当前.git指向的远程仓库地址
git remote rm xxx.git
git remote add xxx.git

git remote remove origin
git remote add origin xxx.git

# 查看git日志
git log

# 查看git状态
git status

# 查看分支
git branch
git branch -avv #看本地
git branch -r # 看远程分支

# 创建并切换到本地分支
git checkout -b 本地分支名 origin/远程分支名
git push orgigin 本地分支名:远程分支名

同步远程分支到本地
git pull origin 远程分支名:本地分支名
其他情况
# 修改commit
# 1、仅commit未push
# 仅修改最近一次commit，且push
git commit --amend
# 修改相关信息 wq保存
# 如果push不成功，可能需要pull同步一下代码，

# 2、修改之前的某次注释
git rebase -i HEAD~2 # 2表示倒数第二次，会显示倒数两次的注释
# 将注释前的pick 修改为 edit 
git commit --amend
git rebase continue

# 3、 仅add未commit
git reset HEAD

# 4、还原本次所有修改，未git add 
git checkout -- * //还原全部文件
git checkout -- aaa.txt //还原指定文件
